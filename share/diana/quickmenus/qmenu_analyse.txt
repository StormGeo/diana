# quickmenu name
"Analyse"

# variables
[UTC= 00,06,12,18]
[Analyse= DNMI,VA,VV,VNN]
[Bakgrunn= white,grayWhite,lightGray,black,orange]
[Kart= darkGreen,white,black]

#
>Bakkeanalyser
FIELD @Analyse.ANA MSLP colour=blue linetype=solid linewidth=2 line.interval=5
OBJECTS NAME="@Analyse Bakkeanalyse" types=front,symbol,area timediff=0
MAP area=modell/sat-område backcolour=@Bakgrunn map=Euro1 cont.colour=@Kart cont.linewidth=1
MAP map=Euro3 cont.colour=@Kart cont.linewidth=1 
LABEL data 
LABEL text="$day $date $auto UTC" tcolour=red bcolour=black fcolour=white:200 polystyle=both halign=left valign=top fontsize=12
#
>Bakkeanalyser med observasjoner
FIELD @Analyse.ANA MSLP colour=blue linetype=solid linewidth=2 line.interval=5
OBS plot=Synop data=Synop,Dribu,Qscat parameter=Vind,TTT,TdTdTd,PPPP,ppp,a,h,VV,N,RRR,ww,W1,W2,Nh,Cl,Cm,Ch,vs,ds,TwTwTw,PwaHwa,Dw1Dw1,Pw1Hw1,TxTn,sss,911ff,s,fxfx,tid,kjtegn  density=1.0 scale=1.0 colour=black timediff=60
OBJECTS NAME="@Analyse Bakkeanalyse" types=front,symbol,area timediff=0
MAP area=modell/sat-område backcolour=@Bakgrunn map=Euro1 cont.colour=@Kart cont.linewidth=1
MAP map=Euro3 colour=@Kart cont.linewidth=1 
LABEL data 
LABEL text="$day $date $auto UTC" tcolour=red bcolour=black fcolour=white:200 polystyle=both halign=left valign=top fontsize=12
